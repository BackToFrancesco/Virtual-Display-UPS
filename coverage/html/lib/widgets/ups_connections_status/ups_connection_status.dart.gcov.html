class UpsConnectionStatusRealTime extends StatelessWidget {
  const UpsConnectionStatusRealTime({Key? key}) : super(key: key);

  @override
  Widget build(BuildContext context) {
    return BlocBuilder<UpsConnectionHandlerBloc, UpsConnectionHandlerState>(
        builder: (context, state) {
      if (state.upsConnectionStatus == UpsConnectionStatus.connected) {
        return _getConnectionStatus(context, app_colors.lightGreen);
      } else if (state.upsConnectionStatus ==
          UpsConnectionStatus.reconnecting) {
        return _getConnectionStatus(context, app_colors.yellow);
      } else {
        return _getConnectionStatus(context, app_colors.lightRed);
      }
    });
  }
